			energy = 6
			mult = planet.value:tec_prime_drone_scaling
		}
	}

	inline_script = jobs/effects/type/inline_evolved_calculator_effect

	inline_script = jobs/effects/type/inline_evolved_tec_prime_drone_effect

	weight = {
		weight = 10000
		modifier = {
			factor = 0.01 # crisis purge
			exists = planet
			exists = planet.controller
			planet.controller = {
				tec_control_crisis_country = yes
			}
		}
	}
}
