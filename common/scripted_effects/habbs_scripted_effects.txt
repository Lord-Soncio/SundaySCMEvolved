habbs_no_void_borne = {
    set_variable = { which = habbs value = 5 }
    every_system_within_border = {
        prev = { change_variable = { which = habbs value = 0.1 } }
    }
    if = {
        limit = {
            check_variable = { which = habbs value > 25 }
        }
        set_variable = { which = habbs value = 25 }
    }
    every_planet_within_border = {
        limit = { is_planet_class = pc_habitat }
        prev = { change_variable = { which = habbs value = -1 } }
    }
}
